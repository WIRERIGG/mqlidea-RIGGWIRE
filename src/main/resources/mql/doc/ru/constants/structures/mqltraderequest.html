MqlTradeRequest

Структура определяет торговый запрос для отправки на сервер с помощью функции OrderSend().

struct MqlTradeRequest
  {
   ENUM_TRADE_REQUEST_ACTIONS action;         // тип торговой операции
   ulong                     magic;          // идентификатор эксперта (магическое число)
   ulong                     order;          // тикет ордера
   string                    symbol;         // символ торговли
   double                    volume;         // запрашиваемый объём в лотах
   double                    price;          // цена
   double                    stoplimit;      // уровень StopLimit ордера
   double                    sl;             // уровень Stop Loss
   double                    tp;             // уровень Take Profit
   ulong                     deviation;      // максимальное отклонение цены
   ENUM_ORDER_TYPE          type;           // тип ордера
   ENUM_ORDER_TYPE_FILLING  type_filling;   // тип исполнения ордера
   ENUM_ORDER_TYPE_TIME     type_time;      // тип истечения ордера
   datetime                 expiration;     // время истечения ордера
   string                    comment;        // комментарий к ордеру
   ulong                     position;       // тикет позиции
   ulong                     position_by;    // тикет встречной позиции
  };

Примечание

Набор полей, которые необходимо заполнить, зависит от типа торговой операции, указанной в поле action. См. TRADE_ACTION для списка возможных торговых действий.
